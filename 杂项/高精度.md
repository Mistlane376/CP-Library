# 高精度

## 非负大整数（基数 1e9）

```cpp
struct BigInt {
    static const int BASE = 1000000000;
    vector<int> d;

    BigInt(long long x = 0) { *this = x; }

    BigInt& operator=(long long x) {
        d.clear();
        if (x == 0) {
            d.push_back(0);
            return *this;
        }
        while (x) {
            d.push_back(x % BASE);
            x /= BASE;
        }
        return *this;
    }
};
```

**算法说明**：
- 使用 `BASE = 1e9` 减少数组长度
- `d[i]` 存储第 i 位的数字（0 ≤ d[i] < BASE）
- 低位存储在数组前面

**常见操作**：
```cpp
// 加法
BigInt operator+(const BigInt& other) const {
    BigInt res;
    res.d.resize(max(d.size(), other.d.size()) + 1);
    int carry = 0;
    for (int i = 0; i < res.d.size(); i++) {
        int sum = carry;
        if (i < d.size()) sum += d[i];
        if (i < other.d.size()) sum += other.d[i];
        res.d[i] = sum % BASE;
        carry = sum / BASE;
    }
    while (res.d.size() > 1 && res.d.back() == 0) res.d.pop_back();
    return res;
}

// 比较
bool operator<(const BigInt& other) const {
    if (d.size() != other.d.size())
        return d.size() < other.d.size();
    for (int i = d.size() - 1; i >= 0; i--)
        if (d[i] != other.d[i])
            return d[i] < other.d[i];
    return false;
}
```
