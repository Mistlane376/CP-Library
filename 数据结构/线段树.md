# 线段树

## 区间求和（支持区间加法）

```cpp
class SegmentTreeSumAdd {
private:
    int n;
    vector<long long> st;
    vector<long long> lazy;

    void build(int p, int L, int R, const vector<int>& A) {
        if (L == R) {
            st[p] = A[L];
            return;
        }
        int m = (L + R) / 2;
        build(p * 2, L, m, A);
        build(p * 2 + 1, m + 1, R, A);
        st[p] = st[p * 2] + st[p * 2 + 1];
    }

    void propagate(int p, int L, int R) {
        if (lazy[p] == 0) return;
        st[p] += lazy[p] * (R - L + 1);
        if (L != R) {
            lazy[p * 2] += lazy[p];
            lazy[p * 2 + 1] += lazy[p];
        }
        lazy[p] = 0;
    }

public:
    SegmentTreeSumAdd(int sz) : n(sz) {
        st.assign(4 * n + 1, 0);
        lazy.assign(4 * n + 1, 0);
    }
};
```

**功能说明**：
- 支持区间加法更新
- 支持区间求和查询
- 使用懒标记（lazy propagation）优化
- 时间复杂度：O(log n) 每次操作

**补充功能**：
```cpp
// 区间更新
void update(int p, int L, int R, int ul, int ur, int val) {
    propagate(p, L, R);
    if (ul > R || ur < L) return;
    if (ul <= L && R <= ur) {
        lazy[p] += val;
        propagate(p, L, R);
        return;
    }
    int m = (L + R) / 2;
    update(p*2, L, m, ul, ur, val);
    update(p*2+1, m+1, R, ul, ur, val);
    st[p] = st[p*2] + st[p*2+1];
}

// 区间查询
long long query(int p, int L, int R, int ql, int qr) {
    propagate(p, L, R);
    if (ql > R || qr < L) return 0;
    if (ql <= L && R <= qr) return st[p];
    int m = (L + R) / 2;
    return query(p*2, L, m, ql, qr) + query(p*2+1, m+1, R, ql, qr);
}
```
